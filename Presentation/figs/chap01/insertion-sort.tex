\begin{algorithm}[H]\alglr
  \caption{Insertion Sort} 
  \begin{algorithmic}[1]
    %\Procedure{Insetion-Sort}{A, n}
    \Proc{Insetion-Sort}{A, n}
    \newline\Comment{A is an array of n elements}
      \For{k \gin \gst{2,...,n}}
        \State key := A[k]
        \State i := k
        \While{ i > 1 \gand A[i-1] > key }
          \State A[i] := A[i-1]
          \State i := i-1
        \EndWhile
        \State A[i] := key
      \EndFor
    %\EndProcedure
  \end{algorithmic}
  \label{alg:insertion-sort}
\end{algorithm}
